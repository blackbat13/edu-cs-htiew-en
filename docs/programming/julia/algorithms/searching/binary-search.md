function binarySearch(array, number)
    left = 1
    right = length(array)

    while left <= right
        middle = (left + right) รท 2

        if number < array[middle]
            right = middle
        elseif number > array[middle]
            left = middle + 1
        else
            return middle
        end
    end

    return -1
end
